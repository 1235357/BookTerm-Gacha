# For NVIDIA GPU users, install PyTorch with CUDA support FIRST:
#
#   1. Check your CUDA version:
#      nvidia-smi
#
#   2. Install PyTorch with matching CUDA version:
#      
#      # For CUDA 12.6 (recommended for latest GPUs):
#      pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu126
#      
#      # For CUDA 12.4:
#      pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu124
#      
#      # For CUDA 12.1:
#      pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu121
#      
#      # For CUDA 11.8:
#      pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118
#      
#      # For CPU only (no GPU):
#      pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cpu
#
#   3. Then install other dependencies:
#      pip install -r requirements.txt
#
#   4. Verify GPU is available:
#      python -c "import torch; print(f'CUDA: {torch.cuda.is_available()}, Device: {torch.cuda.get_device_name(0) if torch.cuda.is_available() else \"CPU\"}')"
